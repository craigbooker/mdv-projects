 34 (function() {
 35     var errors = [],
 36         reference_blocks = [],
 37         processing_time = 0,
 38         regex =
 39         {
 40             "regLevel" : new RegExp("^([\\s\\-]+)"),
 41             "invalidLine" : new RegExp("^\\-\\-\\-|^\\.\\.\\.|^\\s*#.*|^\\s*$"),
 42             "dashesString" : new RegExp("^\\s*\\\"([^\\\"]*)\\\"\\s*$"),
 43             "quotesString" : new RegExp("^\\s*\\\'([^\\\']*)\\\'\\s*$"),
 44             "float" : new RegExp("^[+-]?[0-9]+\\.[0-9]+(e[+-]?[0-9]+(\\.[0-9]+)?)?$"),
 45             "integer" : new RegExp("^[+-]?[0-9]+$"),
 46             "array" : new RegExp("\\[\\s*(.*)\\s*\\]"),
 47             "map" : new RegExp("\\{\\s*(.*)\\s*\\}"),
 48             "key_value" : new RegExp("([a-z0-9_-][ a-z0-9_-]*):( .+)", "i"),
 49             "single_key_value" : new RegExp("^([a-z0-9_-][ a-z0-9_-]*):( .+?)$", "i"),
 50             "key" : new RegExp("([a-z0-9_-][ a-z0-9_-]+):( .+)?", "i"),
 51             "item" : new RegExp("^-\\s+"),
 52             "trim" : new RegExp("^\\s+|\\s+$"),
 53             "comment" : new RegExp("([^\\\'\\\"#]+([\\\'\\\"][^\\\'\\\"]*[\\\'\\\"])*)*(#.*)?")
 54         };

 56      /**
 57       * @class A block of lines of a given level.
 58       * @param {int} lvl The block's level.
 59       * @private
 60       */
 61     function Block(lvl) {
 62         return {
 63             /* The block's parent */
 64             parent: null,
 65             /* Number of children */
 66             length: 0,
 67             /* Block's level */
 68             level: lvl,
 69             /* Lines of code to process */
 70             lines: [],
 71             /* Blocks with greater level */
 72             children : [],
 73             /* Add a block to the children collection */
 74             addChild : function(obj) {
 75                 this.children.push(obj);
 76                 obj.parent = this;
 77                 ++this.length;
 78             }
 79         };
 80     }

 82     // function to create an XMLHttpClient in a cross-browser manner
 83     function createXMLHTTPRequest() {
 84         var xmlhttp;
 85         
 86         try {
 87             // Mozilla / Safari / IE7
 88             xmlhttp = new XMLHttpRequest();
 89         } catch (e) {
 90             // IE
 91             var XMLHTTP_IDS = new Array('MSXML2.XMLHTTP.5.0',
 92             'MSXML2.XMLHTTP.4.0',
 93             'MSXML2.XMLHTTP.3.0',
 94             'MSXML2.XMLHTTP',
 95             'Microsoft.XMLHTTP' );
 96             var success = false;
 97             for (var i=0;i < XMLHTTP_IDS.length && !success; i++) {
 98                 try {
 99                 xmlhttp = new ActiveXObject(XMLHTTP_IDS[i]);
100                 success = true;
101                 } catch (e) {}
102             }
103             if (!success) {
104                 throw new Error('Unable to create XMLHttpRequest.');
105             }
106         }
107         
108         return xmlhttp;
109     }

111     function fromURL(src, ondone) {
112         var client = createXMLHTTPRequest();
113         client.onreadystatechange = function() {
114             if (this.readyState == 4 || this.status == 200) {
115                 var txt = this.responseText;
116                 ondone(YAML.eval(txt));
117             }
118         };
119         client.open('GET', src);
120         client.send();
121     }

123     function parser(str) {
124         var regLevel = regex["regLevel"];
125         var invalidLine = regex["invalidLine"];
126         var lines = str.split("\n");
127         var m;
128         var level = 0, curLevel = 0;
129         
130         var blocks = [];
131         
132         var result = new Block(-1);
133         var currentBlock = new Block(0);
134         result.addChild(currentBlock);
135         var levels = [];
136         var line = "";
137         
138         blocks.push(currentBlock);
139         levels.push(level);
140         
141         for(var i = 0, len = lines.length; i < len; ++i) {
142             line = lines[i];
143             
144             if(line.match(invalidLine)) {
145                 continue;
146             }
147         
148             if(m = regLevel.exec(line)) {
149                 level = m[1].length;
150             } else
151                 level = 0;
152             
153             if(level > curLevel) {
154                 var oldBlock = currentBlock;
155                 currentBlock = new Block(level);
156                 oldBlock.addChild(currentBlock);
157                 blocks.push(currentBlock);
158                 levels.push(level);
159             } else if(level < curLevel) {                
160                 var added = false;
161 
162                 var k = levels.length - 1;
163                 for(; k >= 0; --k) {
164                     if(levels[k] == level) {
165                         currentBlock = new Block(level);
166                         blocks.push(currentBlock);
167                         levels.push(level);
168                         if(blocks[k].parent!= null)
169                             blocks[k].parent.addChild(currentBlock);
170                         added = true;
171                         break;
172                     }
173                 }
174                 
175                 if(!added) {
176                     errors.push("Error: Invalid indentation at line " + i + ": " + line);
177                     return;
178                 }
179             }
180             
181             currentBlock.lines.push(line.replace(regex["trim"], ""));
182             curLevel = level;
183         }
184         
185         return result;
186     }

188     function processValue(val) {
189         val = val.replace(regex["trim"], "");
190         var m = null;
191 
192         if(val == 'true') {
193             return true;
194         } else if(val == 'false') {
195             return false;
196         } else if(val == '.NaN') {
197             return Number.NaN;
198         } else if(val == 'null') {
199             return null;
200         } else if(val == '.inf') {
201             return Number.POSITIVE_INFINITY;
202         } else if(val == '-.inf') {
203             return Number.NEGATIVE_INFINITY;
204         } else if(m = val.match(regex["dashesString"])) {
205             return m[1];
206         } else if(m = val.match(regex["quotesString"])) {
207             return m[1];
208         } else if(m = val.match(regex["float"])) {
209             return parseFloat(m[0]);
210         } else if(m = val.match(regex["integer"])) {
211             return parseInt(m[0]);
212         } else if( !isNaN(m = Date.parse(val))) {
213             return new Date(m);
214         } else if(m = val.match(regex["single_key_value"])) {
215             var res = {};
216             res[m[1]] = processValue(m[2]);
217             return res;
218         } else if(m = val.match(regex["array"])){
219             var count = 0, c = ' ';
220             var res = [];
221             var content = "";
222             var str = false;
223             for(var j = 0, lenJ = m[1].length; j < lenJ; ++j) {
224                 c = m[1][j];
225                 if(c == '\'' || c == '"') {
226                     if(str === false) {
227                         str = c;
228                         content += c;
229                         continue;
230                     } else if((c == '\'' && str == '\'') || (c == '"' && str == '"')) {
231                         str = false;
232                         content += c;
233                         continue;
234                     }
235                 } else if(str === false && (c == '[' || c == '{')) {
236                     ++count;
237                 } else if(str === false && (c == ']' || c == '}')) {
238                     --count;
239                 } else if(str === false && count == 0 && c == ',') {
240                     res.push(processValue(content));
241                     content = "";
242                     continue;
243                 }
244                 
245                 content += c;
246             }
247             
248             if(content.length > 0)
249                 res.push(processValue(content));
250             return res;
251         } else if(m = val.match(regex["map"])){
252             var count = 0, c = ' ';
253             var res = [];
254             var content = "";
255             var str = false;
256             for(var j = 0, lenJ = m[1].length; j < lenJ; ++j) {
257                 c = m[1][j];
258                 if(c == '\'' || c == '"') {
259                     if(str === false) {
260                         str = c;
261                         content += c;
262                         continue;
263                     } else if((c == '\'' && str == '\'') || (c == '"' && str == '"')) {
264                         str = false;
265                         content += c;
266                         continue;
267                     }
268                 } else if(str === false && (c == '[' || c == '{')) {
269                     ++count;
270                 } else if(str === false && (c == ']' || c == '}')) {
271                     --count;
272                 } else if(str === false && count == 0 && c == ',') {
273                     res.push(content);
274                     content = "";
275                     continue;
276                 }
277                 
278                 content += c;
279             }
280             
281             if(content.length > 0)
282                 res.push(content);
283                 
284             var newRes = {};
285             for(var j = 0, lenJ = res.length; j < lenJ; ++j) {
286                 if(m = res[j].match(regex["key_value"])) {
287                     newRes[m[1]] = processValue(m[2]);
288                 }
289             }
290             
291             return newRes;
292         } else 
293             return val;
294     }

296     function processFoldedBlock(block) {
297         var lines = block.lines;
298         var children = block.children;
299         var str = lines.join(" ");
300         var chunks = [str];
301         for(var i = 0, len = children.length; i < len; ++i) {
302             chunks.push(processFoldedBlock(children[i]));
303         }
304         return chunks.join("\n");
305     }

307     function processLiteralBlock(block) {
308         var lines = block.lines;
309         var children = block.children;
310         var str = lines.join("\n");
311         for(var i = 0, len = children.length; i < len; ++i) {
312             str += processLiteralBlock(children[i]);
313         }
314         return str;
315     }

317     function processBlock(blocks) {
318         var m = null;
319         var res = {};
320         var lines = null;
321         var children = null;
322         var currentObj = null;
323         
324         var level = -1;
325         
326         var processedBlocks = [];
327         
328         var isMap = true;
329         
330         for(var j = 0, lenJ = blocks.length; j < lenJ; ++j) {
331             
332             if(level != -1 && level != blocks[j].level)
333                 continue;
334         
335             processedBlocks.push(j);
336         
337             level = blocks[j].level;
338             lines = blocks[j].lines;
339             children = blocks[j].children;
340             currentObj = null;
341         
342             for(var i = 0, len = lines.length; i < len; ++i) {
343                 var line = lines[i];
344 
345                 if(m = line.match(regex["key"])) {
346                     var key = m[1];
347                     
348                     if(key[0] == '-') {
349                         key = key.replace(regex["item"], "");
350                         if (isMap) { 
351                             isMap = false;
352                             if (typeof(res.length) === "undefined") {
353                                 res = [];
354                             } 
355                         }
356                         if(currentObj != null) res.push(currentObj);
357                         currentObj = {};
358                         isMap = true;
359                     }
360                     
361                     if(typeof m[2] != "undefined") {
362                         var value = m[2].replace(regex["trim"], "");
363                         if(value[0] == '&') {
364                             var nb = processBlock(children);
365                             if(currentObj != null) currentObj[key] = nb;
366                             else res[key] = nb;
367                             reference_blocks[value.substr(1)] = nb;
368                         } else if(value[0] == '|') {
369                             if(currentObj != null) currentObj[key] = processLiteralBlock(children.shift());
370                             else res[key] = processLiteralBlock(children.shift());
371                         } else if(value[0] == '*') {
372                             var v = value.substr(1);
373                             var no = {};
374                             
375                             if(typeof reference_blocks[v] == "undefined") {
376                                 errors.push("Reference '" + v + "' not found!");
377                             } else {
378                                 for(var k in reference_blocks[v]) {
379                                     no[k] = reference_blocks[v][k];
380                                 }
381                                 
382                                 if(currentObj != null) currentObj[key] = no;
383                                 else res[key] = no;
384                             }
385                         } else if(value[0] == '>') {
386                             if(currentObj != null) currentObj[key] = processFoldedBlock(children.shift());
387                             else res[key] = processFoldedBlock(children.shift());
388                         } else {
389                             if(currentObj != null) currentObj[key] = processValue(value);
390                             else res[key] = processValue(value);
391                         }
392                     } else {
393                         if(currentObj != null) currentObj[key] = processBlock(children);
394                         else res[key] = processBlock(children);                        
395                     }
396                 } else if(line.match(/^-\s*$/)) {
397                     if (isMap) { 
398                         isMap = false;
399                         if (typeof(res.length) === "undefined") {
400                             res = [];
401                         } 
402                     }
403                     if(currentObj != null) res.push(currentObj);
404                     currentObj = {};
405                     isMap = true;
406                     continue;
407                 } else if(m = line.match(/^-\s*(.*)/)) {
408                     if(currentObj != null) 
409                         currentObj.push(processValue(m[1]));
410                     else {
411                         if (isMap) { 
412                             isMap = false;
413                             if (typeof(res.length) === "undefined") {
414                                 res = [];
415                             } 
416                         }
417                         res.push(processValue(m[1]));
418                     }
419                     continue;
420                 }
421             }
422             
423             if(currentObj != null) {
424                 if (isMap) { 
425                     isMap = false;
426                     if (typeof(res.length) === "undefined") {
427                         res = [];
428                     } 
429                 }
430                 res.push(currentObj);
431             }
432         }
433         
434         for(var j = processedBlocks.length - 1; j >= 0; --j) {
435             blocks.splice.call(blocks, processedBlocks[j], 1);
436         }
437 
438         return res;
439     }

441     function semanticAnalysis(blocks) {
442         var res = processBlock(blocks.children);
443         return res;
444     }

446     function preProcess(src) {
447         var m;
448         var lines = src.split("\n");
449         
450         var r = regex["comment"];
451         
452         for(var i in lines) {
453             if(m = lines[i].match(r)) {
454 /*                var cmt = "";
455                 if(typeof m[3] != "undefined")
456                     lines[i] = m[1];
457                 else if(typeof m[3] != "undefined")
458                     lines[i] = m[3]; 
459                 else
460                     lines[i] = "";
461                     */
462                 if(typeof m[3] !== "undefined") {
463                     lines[i] = m[0].substr(0, m[0].length - m[3].length);
464                 }
465             }
466         }
467         
468         return lines.join("\n");
469     }

471     function eval(str) {
472         errors = [];
473         reference_blocks = [];
474         processing_time = (new Date()).getTime();
475         var pre = preProcess(str)
476         var doc = parser(pre);
477         var res = semanticAnalysis(doc);
478         processing_time = (new Date()).getTime() - processing_time;
479         
480         return res;
481     }
